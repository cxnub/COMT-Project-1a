 88    def update_character_stats(self, character: BaseCharacter):
 89        """Update character statistics.
 90
 91        Parameters
 92        ----------
 93        character : BaseCharacter
 94            The character instance.
 95        """
 96
 97        # update character name
 98        name = ("You " if character.is_player() else "Enemy ") + f"({character.name})"
 99        self.configure(text=name)
100
101        # update hp
102        self.dynamic_hp.set(character.health_points)
103        self.dynamic_max_hp.set(character.max_health_points)
104        self.dynamic_hp_text.set(f"{character.health_points}/{character.max_health_points}")
105
106        # update dp
107        self.dynamic_dp.set(character.defense_points)
108        self.dynamic_max_dp.set(character.max_defense_points)
109        self.dynamic_dp_text.set(f"{character.defense_points}/{character.max_defense_points}")
110
111        # update AP, SP, Luck
112        self.dynamic_ap.set(f"AP: {character.attack_points} points")
113        self.dynamic_sp.set(f"SP: {character.speed_points} points")
114        self.dynamic_luck.set(f"Luck: {character.luck} points")
115
116        if character.is_player():
117            # update MP and active effects if character is player
118            self.dynamic_mp.set(f"MP: {character.magic_points} points")
119            active_effect_names = [effect.name for effect in character.active_effects]
120            self.dynamic_active_effects.set(f"Active Effects: {active_effect_names}")

172    def create_stats_progress_bar(
173        self, text, int_variable: tk.Variable, text_variable: tk.Variable, value: int,
174        max_value: int, row: int
175        ):
176        """
177        Create and configure a progress bar for a specific character statistic.
178
179        Parameters
180        ----------
181        text : str
182            The label text for the statistic.
183        int_variable : tk.Variable
184            The Tkinter IntVar variable holding the current value of the statistic.
185        text_variable : tk.Variable
186            The Tkinter StringVar variable holding the text displaying the current value
187            of the statistic along with the maximum value.
188        value : int
189            The current value of the statistic.
190        max_value : int
191            The maximum value of the statistic.
192        row : int
193            The grid row where the progress bar will be placed.
194        """
195
196        # create progress bar frame
197        progress_bar_frame = tk.Frame(self)
198
199        # configure stats label
200        stat_label = ttk.Label(progress_bar_frame, text=text)
201        stat_label.grid(column=0, row=row)
202
203        # configure progrss bar
204        progress_bar = ttk.Progressbar(progress_bar_frame)
205        progress_bar.configure(
206            maximum=max_value,
207            variable=int_variable
208            )
209        progress_bar.grid(column=1, row=row, padx=5)
210
211        # configure value label
212        value_label_text = f"{value}/{max_value}"
213        value_label = ttk.Label(
214            progress_bar_frame, text=value_label_text, textvariable=text_variable
215            )
216        value_label.grid(column=2, row=row)
217
218        # configure progress_bar_frame
219        progress_bar_frame.grid(row=row, column=0, sticky="w")
220
221        # add to lists to avoid garbage collected
222        self.widgets.append(progress_bar_frame)
223        self.progress_bars.append(progress_bar)
224
225    def create_stats_widgets(self, character: BaseCharacter):
226        """Create and configure various character statistics widgets.
227        
228        Parameters
229        ----------
230        character : BaseCharacter
231            The character instance.
232        """
233
234        # create image
235        self.create_image(character)
236
237        # hp and dp parameters for self.create_stats_progress_bar
238        progress_bar_parameters = [
239            (
240                "HP:", self.dynamic_hp, self.dynamic_hp_text,
241                character.health_points, character.max_health_points
242                ),
243
244            (
245                "DP:", self.dynamic_dp, self.dynamic_dp_text,
246                character.defense_points, character.max_defense_points
247                )
248        ]
249
250        # create hp and dp progress bars
251        for index, parameters in enumerate(progress_bar_parameters, start=2):
252            self.create_stats_progress_bar(*parameters, index)
253
254        # list of stats label text
255        stats_labels_text = [
256            (f"AP: {character.attack_points} points", self.dynamic_ap),
257            (f"SP: {character.speed_points} points", self.dynamic_sp),
258            (f"Luck: {character.luck} points", self.dynamic_luck)
259        ]
260
261        # add magic point stat if character is a player
262        if character.is_player():
263            active_effect_names = [effect.name for effect in character.active_effects]
264            stats_labels_text.extend(
265                    [
266                    (f"MP: {character.magic_points} points", self.dynamic_mp),
267                    (f"Active Effects: {active_effect_names}", self.dynamic_active_effects)
268                    ]
269                )
270
271        # create labels for displaying character statistics
272        for row, items in enumerate(stats_labels_text, start=4):
273            # extract the stat_text and text_variable from the items tuple
274            stat_text, text_variable = items
275
276            # configure stat label
277            stat_label = ttk.Label(self)
278            stat_label.configure(text=stat_text, textvariable=text_variable)
279            stat_label.grid(column=0, row=row, sticky="w")
280
281            # add to widget lists to avoid garbage collected
282            self.widgets.append(stat_label)
