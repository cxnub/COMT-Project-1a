 81    def game_loop(self):
 82        """Function for the game loop."""
 83
 84        winner = self.game_manager.is_game_ended()
 85
 86        # handle game over
 87        if winner == "player":
 88            self.add_battle_log("You won!")
 89            self.handle_game_over()
 90            return
 91
 92        elif winner == "enemy":
 93            self.add_battle_log("You lost.")
 94            self.handle_game_over()
 95            return
 96
 97        # update character state
 98        self.check_characters_health()
 99
100        if not self.player_thinking:
101            self.turn_character = self.game_manager.determine_turn_order()
102
103        if self.turn_character.is_player():
104            self.player_thinking = True
105
106        else:
107            # temporarily disable action buttons
108            self.configure_action_buttons_state("disabled")
109
110            # choose enemy action after 500ms
111            self.after(1000, self.run_enemy_turn, self.turn_character)
112
113            # enable actions buttons
114            self.after(1000, self.configure_action_buttons_state, "enabled")
115
116            # run game loop after 500ms
117            self.after(1000, self.game_loop)
118
119        # set turn label text
120        self.turn_label_text.set(f"It's {self.turn_character.name}'s turn.")

313    def check_characters_health(self):
314        """Check if any character is defeated and handles it."""
315
316        player = self.game_manager.active_player
317        enemy = self.game_manager.active_enemy
318
319        # checks if any character died
320        if not player.is_alive():
321            # handle defeated active player
322            self.game_manager.switch_player_character()
323            self.add_battle_log(f"{player.name} has been defeated by {enemy.name}!")
324            self.update_ui()
325
326        if not enemy.is_alive():
327            # handle defeated active enemy
328            self.game_manager.switch_enemy_character()
329            self.add_battle_log(f"{enemy.name} has been defeated by {player.name}!")
330            self.update_ui()

332    def update_ui(self):
333        """Update the whole combat frame UI."""
334
335        # store character frames and instances in a list
336        character_frames = [self.player_label_frame, self.enemy_label_frame]
337        character_instances = [self.game_manager.active_player, self.game_manager.active_enemy]
338
339        # zip frames and instances and loop through
340        for frame, character in zip(character_frames, character_instances):
341
342            # update character stats
343            frame.update_frame(character)
344
345        # update skill buttons
346        player_skills = self.game_manager.active_player.skills
347        self.skill_one_text.set(player_skills[0].name)
348        self.skill_two_text.set(player_skills[1].name)

350    def add_battle_log(self, log: str):
351        """Adds a battle log with time prefix.
352        
353        Parameters
354        ----------
355        battle_log : str
356            The log to add.
357        """
358
359        # get current time
360        current_time = datetime.now().strftime("%H:%M:%S - ")
361
362        # enable text widget
363        self.battle_log_text_widget.configure(state="normal")
364
365        # add current time to log
366        log = current_time + log
367
368        # makes sure log fits in frame
369        log = "\n".join(split_text(log, 65))
370
371        # add log to battle log text widget
372        self.battle_log_text_widget.insert("end", log + "\n")
373
374        # scroll to the end
375        self.battle_log_text_widget.see("end")
376
377        # disable text widget
378        self.battle_log_text_widget.configure(state="disabled")

381    def on_action_pressed(self, action: str):
382        """Handles action press.
383        
384        action : str
385            The action pressed.
386        """
387
388        # block action if enemy is thinking
389        if not self.player_thinking:
390            return
391
392        # handle switch character pressed
393        if action.lower() == "switch character":
394
395            # shows the switch character frame
396            self.master.switch_character_frame = SwitchCharacterFrame(
397                self.master, self.game_manager.player_characters
398                )
399            self.master.show_frame(self.master.switch_character_frame)
400
401            # switch character and log it
402            log = f"Character switched to {self.game_manager.active_player.name}!"
403
404        # handle attack button pressed
405        elif action.lower() == "attack":
406
407            # use attack and get the log
408            log = self.game_manager.player_attack()
409
410        # handle defend button pressed
411        elif action.lower() == "defend":
412
413            # use defend and get the log
414            log = self.game_manager.player_defend()
415
416        # handle skill button pressed
417        else:
418
419            # names of player skills
420            player_skills_names = [
421                skill.name for skill in self.game_manager.active_player.skills
422                ]
423
424            # use skill and get the log
425            log = self.game_manager.player_use_skill(player_skills_names.index(action))
426
427        # update idle character stats
428        self.game_manager.update_idle_character_stats(self.game_manager.active_enemy)
429
430        # set player_thinking to False, except when not enough points for Skill
431        if not log.startswith("Not enough"):
432            self.player_thinking = False
433
434        # update battle log
435        self.add_battle_log(log)
436
437        # update UI
438        self.update_ui()
439
440        # run game loop
441        self.game_loop()
442
