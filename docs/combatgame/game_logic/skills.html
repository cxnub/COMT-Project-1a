144        def take_effect(self, attacker: "BaseCharacter", damage: int):
145            """Use the effect.
146            
147            Parameters
148            ----------
149            attacker : BaseCharacter
150                The attacker that tried to attack.
151            damage : int
152                The damage the attacker intended to deal.
153
154            Returns
155            -------
156            log : str
157                The battle log.
158            """
159
160            # if damage more than attacker's defense points,
161            # deal remaining damage to attacker's health points
162            defense_points_damage = min(damage, attacker.defense_points)
163            health_points_damage = max(0, damage - attacker.defense_points)
164
165            # deals the damage
166            attacker.defense_points -= defense_points_damage
167            attacker.health_points -= health_points_damage
168
169            log = f"{attacker.name}'s attack was met with a defensive shield, " \
170                "causing the damage to reflect back to themselves. " \
171                f"(-{defense_points_damage}DP and -{health_points_damage}HP)"
172
173            return log

176            return self.name
