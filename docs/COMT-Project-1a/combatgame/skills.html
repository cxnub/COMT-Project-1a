 25class BaseSkill:
 26    """Represents a skill.

 39    speed_points_cost : int
 40        The amount of speed points cost to use this skill.
 41
 42    require_target : bool
 43        True if skill affects an enemy, False otherwise.
 44
 45    message_displays : list of str
 46        A list of message displays when skill is used.
 47
 48    belongs_to : str
 49        The character the skill belongs to.
 50    """
 51
 52    name: str = ""
 53    description: str = ""
 54    magic_points_cost: int = 0
 55    speed_points_cost: int = 0
 56    require_target: bool = False
 57    message_displays: list[str] = []
 58    belongs_to: str = ""

 60    def __init__(self, skill_class_name: str):
 61        """Initialize a skill instance.
 62
 63        Parameters
 64        ----------
 65        skill_class_name : str
 66            The name of the class of the skill.
 67        """
 68        self._assign_skill_attributes(skill_class_name)

 70    def _assign_skill_attributes(self, skill_class_name: str):
 71        """Assign skill attributes from skill_attributes.csv.
 72
 73        Parameters
 74        ----------
 75        skill_class_name : str
 76            The name of the class of the skill.
 77        """
 78        attr = skill_attributes[skill_class_name]
 79        self.name: str = str(attr["name"])
 80        self.magic_points_cost: int = int(attr["mp_cost"])
 81        self.speed_points_cost: int = int(attr["sp_cost"])
 82        require_target_attr = str(attr["require_target"]).lower()
 83        self.require_target: bool = require_target_attr == "yes"
 84        self.belongs_to: str = str(attr["belongs_to"])

139        def take_effect(self, attacker: "BaseCharacter", damage: int):
140            """Use the effect.
141            
142            Parameters
143            ----------
144            attacker : BaseCharacter
145                The attacker that tried to attack.
146            damage : int
147                The damage the attacker intended to deal.
148
149            Returns
150            -------
151            log : str
152                The battle log.
153            """
154
155            # if damage more than attacker's defense points,
156            # deal remaining damage to attacker's health points
157            defense_points_damage = min(damage, attacker.defense_points)
158            health_points_damage = max(0, damage - attacker.defense_points)
159
160            # deals the damage
161            attacker.defense_points -= defense_points_damage
162            attacker.health_points -= health_points_damage
163
164            log = f"{attacker.name}'s attack was met with a defensive shield, " \
165                "causing the damage to reflect back to themselves. " \
166                f"(-{defense_points_damage}DP and -{health_points_damage}HP)"
167
168            return log
169
170        def __str__(self):
171            return self.name
